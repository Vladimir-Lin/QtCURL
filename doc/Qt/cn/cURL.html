编译libSSH2
















Linux 32 / 64

./configure --disable-shared --enable-static --enable-optimize --enable-ipv6 --disable-curldebug --without-winssl --with-ssl=/Wherever/deps --with-zlib=/Wherever/deps --with-libssh2=/Wherever/deps
make


Mac OS X Clang 64

./configure --disable-shared --enable-static --enable-optimize --enable-ipv6 --disable-curldebug --without-winssl --with-ssl=/Wherever/deps --with-zlib=/Whereverdeps --with-libssh2=/Wherever/deps
make

Windows 7





准备OpenSSL





Windows 32

nmake /f Makefile.vc mode=dll VC=12 WITH_SSL=dll WITH_ZLIB=dll WITH_SSH2=dll DEBUG=no ENABLE_WINSSL=no MACHINE=x86


Windows 64

nmake /f Makefile.vc mode=dll VC=12 WITH_SSL=dll WITH_ZLIB=dll WITH_SSH2=dll ENABLE_IPV6=yes ENABLE_SSPI=yes ENABLE_IDN=yes DEBUG=no ENABLE_WINSSL=no MACHINE=x64


cmake -G "Visual Studio 12 Win64" ..\curl-7.48.0 -DCMAKE_INSTALL_PREFIX=H:\libcURL\x64\shared\release -DCMAKE_PREFIX_PATH=G:\Qt\3rdparty\libssh2\x64\shared\release
cmake -G "Visual Studio 12 Win64" ..\curl-7.48.0 -DENABLE_DEBUG=1 -DCMAKE_INSTALL_PREFIX=H:\libcURL\x64\shared\debug -DCMAKE_PREFIX_PATH=G:\Qt\3rdparty\libssh2\x64\shared\debug
cmake -G "Visual Studio 12 Win64" ..\curl-7.48.0 -DCURL_STATICLIB=1 -DCMAKE_INSTALL_PREFIX=H:\libcURL\x64\static\release -DCMAKE_PREFIX_PATH=G:\Qt\3rdparty\libssh2\x64\static\release
cmake -G "Visual Studio 12 Win64" ..\curl-7.48.0 -DCURL_STATICLIB=1 -DENABLE_DEBUG=1 -DCMAKE_INSTALL_PREFIX=H:\libcURL\x64\static\debug -DCMAKE_PREFIX_PATH=G:\Qt\3rdparty\libssh2\x64\static\debug
